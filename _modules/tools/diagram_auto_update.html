"""–ú–æ–¥—É–ª—å –¥–ª—è –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–≥–æ –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è –¥–∏–∞–≥—Ä–∞–º–º—ã –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–µ–π.

–°–∫—Ä–∏–ø—Ç –∞–Ω–∞–ª–∏–∑–∏—Ä—É–µ—Ç –Ω–æ–≤—ã–µ —Ñ–∞–π–ª—ã –≤ —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏–∏ –∏ –≥–µ–Ω–µ—Ä–∏—Ä—É–µ—Ç PlantUML –¥–∏–∞–≥—Ä–∞–º–º—É,
–æ—Ç–æ–±—Ä–∞–∂–∞—é—â—É—é —Å—Ç—Ä—É–∫—Ç—É—Ä—É –ø—Ä–æ–µ–∫—Ç–∞ —Å –≤–æ–∑–º–æ–∂–Ω–æ—Å—Ç—å—é –ø–µ—Ä–µ—Ö–æ–¥–∞ –∫ —Ñ–∞–π–ª–∞–º —á–µ—Ä–µ–∑ vscode:// —Å—Å—ã–ª–∫–∏.
"""

import subprocess  # nosec B404
from pathlib import Path

# –ü—É—Ç–∏ –∫ –∫–æ—Ä–Ω–µ–≤–æ–π –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏ –∏—Å—Ö–æ–¥–Ω–æ–≥–æ –∫–æ–¥–∞ –∏ –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏ —Å –¥–∏–∞–≥—Ä–∞–º–º–∞–º–∏
BASE_DIR = Path(__file__).parent.parent.resolve()
SRC_DIR = BASE_DIR / "src"
DIAGRAMS_DIR = BASE_DIR / "diagrams"
OUTPUT_FILE = DIAGRAMS_DIR / "temp_containers.puml"

# –ù–∞—Å—Ç—Ä–æ–π–∫–∏ –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è –¥–ª—è —Ñ–∞–π–ª–æ–≤ –ø–æ —Ä–∞—Å—à–∏—Ä–µ–Ω–∏—è–º:
# (–ß–µ–ª–æ–≤–µ—á–µ—Å–∫–æ–µ –∏–º—è —Ç–∏–ø–∞ —Ñ–∞–π–ª–∞, –∏–º—è —Å–ø—Ä–∞–π—Ç–∞ PlantUML, —Ç–µ–≥ –≤ –¥–∏–∞–≥—Ä–∞–º–º–µ)
EXT_SETTINGS = {
    ".py": ("Python", "python", "code"),
    ".js": ("JavaScript", "javascript", "code"),
    ".ts": ("TypeScript", "typescript", "code"),
    ".json": ("JSON", "json", "config"),
    ".yaml": ("YAML", "yaml", "config"),
    ".yml": ("YAML", "yaml", "config"),
    ".txt": ("Text", "text", "docs"),
    ".md": ("Markdown", "markdown", "docs"),
    ".ipynb": ("Notebook", "jupyter", "notebook"),
    ".html": ("HTML", "html", "web"),
    ".css": ("CSS", "css", "web"),
}




[–¥–æ–∫—É–º–µ–Ω—Ç–∞—Ü–∏—è]
def get_changed_files():
    """–ü–æ–ª—É—á–∞–µ—Ç —Å–ø–∏—Å–æ–∫ –Ω–æ–≤—ã—Ö (–Ω–µ–æ—Ç—Å–ª–µ–∂–∏–≤–∞–µ–º—ã—Ö) —Ñ–∞–π–ª–æ–≤ –≤ —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏–∏ Git.

    Returns:
        List[str]: –°–ø–∏—Å–æ–∫ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω—ã—Ö –ø—É—Ç–µ–π –∫ –Ω–æ–≤—ã–º —Ñ–∞–π–ª–∞–º.

    Note:
        –ò—Å–ø–æ–ª—å–∑—É–µ—Ç `git ls-files --others --exclude-standard` –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è —Å–ø–∏—Å–∫–∞ —Ñ–∞–π–ª–æ–≤.
        –í–∞–ª–∏–¥–∞—Ü–∏—è –ø—É—Ç–∏ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è –≤ –≤—ã–∑—ã–≤–∞—é—â–µ–º –∫–æ–¥–µ.
    """
    try:
        result = subprocess.run(  # nosec B603, B607
            ["git", "ls-files", "--others", "--exclude-standard"],
            stdout=subprocess.PIPE,
            text=True,
            cwd=BASE_DIR,
            check=True,
        )
        files = [
            line.strip() for line in result.stdout.strip().split("\n") if line.strip()
        ]
        return files
    except subprocess.CalledProcessError as e:
        print(f"Error getting changed files: {e}")
        return []






[–¥–æ–∫—É–º–µ–Ω—Ç–∞—Ü–∏—è]
def build_container_definitions(added_files):
    """–û–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ—Ç —Å–ø–∏—Å–æ–∫ –¥–æ–±–∞–≤–ª–µ–Ω–Ω—ã—Ö —Ñ–∞–π–ª–æ–≤ –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è –¥–ª—è –¥–∏–∞–≥—Ä–∞–º–º—ã.

    Args:
        added_files (List[str]): –°–ø–∏—Å–æ–∫ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω—ã—Ö –ø—É—Ç–µ–π –∫ –¥–æ–±–∞–≤–ª–µ–Ω–Ω—ã–º —Ñ–∞–π–ª–∞–º.

    Returns:
        Tuple[List[str], Dict]: –ö–æ—Ä—Ç–µ–∂ —Å–æ–¥–µ—Ä–∂–∞—â–∏–π:
            - –°–ø–∏—Å–æ–∫ —Å—Ç—Ä–æ–∫ —Å –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è–º–∏ vscode-—Å—Å—ã–ª–æ–∫
            - –°–ª–æ–≤–∞—Ä—å —Å –∏–µ—Ä–∞—Ä—Ö–∏–µ–π –ø–∞–ø–æ–∫ –∏ —Ñ–∞–π–ª–æ–≤ –¥–ª—è –≥–µ–Ω–µ—Ä–∞—Ü–∏–∏ PlantUML
    """
    defines = []
    boundaries = {}

    for filepath in added_files:
        full_path = BASE_DIR / filepath

        # –§–∏–ª—å—Ç—Ä—É–µ–º —Ç–æ–ª—å–∫–æ —Ñ–∞–π–ª—ã –≤–Ω—É—Ç—Ä–∏ src, –∫–æ—Ç–æ—Ä—ã–µ —Ä–µ–∞–ª—å–Ω–æ —Å—É—â–µ—Å—Ç–≤—É—é—Ç
        if not filepath.startswith("src/") or not full_path.exists():
            continue

        # –ò–≥–Ω–æ—Ä–∏—Ä—É–µ–º —Å–ª—É–∂–µ–±–Ω—ã–µ —Ñ–∞–π–ª—ã Python
        if full_path.name == "__init__.py":
            continue

        ext = full_path.suffix.lower()
        file_type, sprite, tag = EXT_SETTINGS.get(ext, ("File", "file", "other"))

        # –ü–æ–ª—É—á–∞–µ–º –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω—ã–π –ø—É—Ç—å –æ—Ç –∫–æ—Ä–Ω—è –ø—Ä–æ–µ–∫—Ç–∞
        rel_path = full_path.relative_to(BASE_DIR)
        parts = rel_path.parts  # ('src', 'subdir', 'file.py')

        # –§–æ—Ä–º–∏—Ä—É–µ–º —É–Ω–∏–∫–∞–ª—å–Ω—ã–π –∏–¥–µ–Ω—Ç–∏—Ñ–∏–∫–∞—Ç–æ—Ä –∏ vscode-—Å—Å—ã–ª–∫—É (–±–µ–∑ –∂—ë—Å—Ç–∫–æ–≥–æ C:/)
        var_name = (full_path.stem + "_" + ext.lstrip(".")).lower()
        var_const = var_name.upper()
        vscode_link = f"vscode://file/{rel_path.as_posix()}"
        defines.append(f'!define {var_const} "{vscode_link}"')

        # –°—Ç—Ä–æ–∏–º –∏–µ—Ä–∞—Ä—Ö–∏—é –ø–∞–ø–æ–∫: src/.../–ø–∞–ø–∫–∞/—Ñ–∞–π–ª
        current = boundaries
        for part in parts[1:-1]:  # –ø—Ä–æ–ø—É—Å–∫–∞–µ–º "src", –∏–¥—ë–º –ø–æ –ø–∞–ø–∫–∞–º
            current = current.setdefault(part, {})
        current.setdefault("files", []).append(
            (parts[-1], var_name, file_type, sprite, tag, var_const)
        )

    return defines, boundaries






[–¥–æ–∫—É–º–µ–Ω—Ç–∞—Ü–∏—è]
def write_container_block(f, name, content, indent=0):
    """–†–µ–∫—É—Ä—Å–∏–≤–Ω–æ –∑–∞–ø–∏—Å—ã–≤–∞–µ—Ç –±–ª–æ–∫ Container_Boundary –∏ —Ñ–∞–π–ª—ã –≤–Ω—É—Ç—Ä–∏ –Ω–µ–≥–æ.

    Args:
        f: –§–∞–π–ª–æ–≤—ã–π –æ–±—ä–µ–∫—Ç –¥–ª—è –∑–∞–ø–∏—Å–∏.
        name (str): –ò–º—è —Ç–µ–∫—É—â–µ–≥–æ –∫–æ–Ω—Ç–µ–π–Ω–µ—Ä–∞.
        content (Dict): –°–æ–¥–µ—Ä–∂–∏–º–æ–µ –∫–æ–Ω—Ç–µ–π–Ω–µ—Ä–∞ (–ø–æ–¥–ø–∞–ø–∫–∏ –∏ —Ñ–∞–π–ª—ã).
        indent (int): –£—Ä–æ–≤–µ–Ω—å –æ—Ç—Å—Ç—É–ø–∞ –¥–ª—è —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏—è.
    """
    space = " " * indent

    # –ó–∞–ø–∏—Å—ã–≤–∞–µ–º —Ñ–∞–π–ª—ã –≤ —Ç–µ–∫—É—â–µ–π –ø–∞–ø–∫–µ
    if "files" in content:
        for filename, var_name, file_type, sprite, tag, var_const in content["files"]:
            f.write(
                f'{space}    Container({var_name}, "{filename}", "{file_type}", '
                f'"{filename}", $tags="{tag}", $sprite="{sprite}", '
                f'$link="{var_const}")\n'
            )

    # –†–µ–∫—É—Ä—Å–∏–≤–Ω–æ –æ–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ–º –ø–æ–¥–ø–∞–ø–∫–∏
    for subfolder, subcontent in content.items():
        if subfolder == "files":
            continue
        f.write(
            f'{space}    Container_Boundary({subfolder}, "{subfolder}", $tags="code") {{\n'
        )
        write_container_block(f, subfolder, subcontent, indent + 4)
        f.write(f"{space}    }}\n")






[–¥–æ–∫—É–º–µ–Ω—Ç–∞—Ü–∏—è]
def write_temp_puml(defines, boundaries):
    """–ì–µ–Ω–µ—Ä–∏—Ä—É–µ—Ç –≤—Ä–µ–º–µ–Ω–Ω—ã–π .puml —Ñ–∞–π–ª —Å –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è–º–∏ —Å—Å—ã–ª–æ–∫ –∏ –∏–µ—Ä–∞—Ä—Ö–∏–µ–π –∫–æ–º–ø–æ–Ω–µ–Ω—Ç–æ–≤.

    Args:
        defines (List[str]): –°–ø–∏—Å–æ–∫ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–π vscode-—Å—Å—ã–ª–æ–∫.
        boundaries (Dict): –ò–µ—Ä–∞—Ä—Ö–∏—è –ø–∞–ø–æ–∫ –∏ —Ñ–∞–π–ª–æ–≤ –ø—Ä–æ–µ–∫—Ç–∞.
    """
    DIAGRAMS_DIR.mkdir(exist_ok=True)  # –°–æ–∑–¥–∞—ë–º –ø–∞–ø–∫—É, –µ—Å–ª–∏ –Ω–µ—Ç

    with open(OUTPUT_FILE, "w", encoding="utf-8") as f:
        f.write("@startuml\n\n")

        # –í—Å—Ç–∞–≤–ª—è–µ–º –≤—Å–µ !define —Å—Å—ã–ª–∫–∏
        for line in defines:
            f.write(line + "\n")

        f.write('\nContainer_Boundary(src, "src (–∏—Å—Ö–æ–¥–Ω—ã–π –∫–æ–¥)", $tags="code") {\n')
        write_container_block(f, "src", boundaries, indent=4)
        f.write("}\n\n@enduml\n")




if __name__ == "__main__":
    """–û—Å–Ω–æ–≤–Ω–∞—è —Ç–æ—á–∫–∞ –≤—Ö–æ–¥–∞ —Å–∫—Ä–∏–ø—Ç–∞."""
    added_files = get_changed_files()
    print("Changed files detected:", added_files)  # –î–ª—è –æ—Ç–ª–∞–¥–∫–∏

    defines, boundaries = build_container_definitions(added_files)
    write_temp_puml(defines, boundaries)

    print(f"Generated PlantUML diagram at {OUTPUT_FILE}")
