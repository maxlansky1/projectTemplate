"""–ú–æ–¥—É–ª—å –¥–ª—è –Ω–∞—Å—Ç—Ä–æ–π–∫–∏ –∫–∞—Å—Ç–æ–º–Ω–æ–≥–æ –ª–æ–≥–≥–µ—Ä–∞.

–û–±–µ—Å–ø–µ—á–∏–≤–∞–µ—Ç —Ü–≤–µ—Ç–Ω–æ–π –≤—ã–≤–æ–¥ –≤ –∫–æ–Ω—Å–æ–ª—å –∏ –∑–∞–ø–∏—Å—å –ª–æ–≥–æ–≤ –≤ —Ñ–∞–π–ª—ã.

    LOGS_DIR (str): –ü—É—Ç—å –∫ –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏, –≥–¥–µ –±—É–¥—É—Ç —Å–æ—Ö—Ä–∞–Ω—è—Ç—å—Å—è –ª–æ–≥-—Ñ–∞–π–ª—ã.
    LOG_FORMAT (str): –§–æ—Ä–º–∞—Ç —Å—Ç—Ä–æ–∫–∏ –ª–æ–≥–∞.
    DATE_FORMAT (str): –§–æ—Ä–º–∞—Ç –¥–∞—Ç—ã –∏ –≤—Ä–µ–º–µ–Ω–∏ –≤ –ª–æ–≥–∞—Ö.

import logging
import os
from datetime import datetime

from colorama import Back, Fore, Style, init

from configs.config import Config

# –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è colorama –¥–ª—è Windows/Linux/macOS
init(autoreset=True)

# –ü—É—Ç—å –∫ –ø–∞–ø–∫–µ —Å –ª–æ–≥–∞–º–∏
LOGS_DIR = "logs"
os.makedirs(LOGS_DIR, exist_ok=True)

# –§–æ—Ä–º–∞—Ç –ª–æ–≥–æ–≤
LOG_FORMAT = "%(asctime)s | %(levelname)-8s | %(name)s:%(lineno)d | %(message)s"
DATE_FORMAT = "%Y-%m-%d %H:%M:%S"




[–¥–æ–∫—É–º–µ–Ω—Ç–∞—Ü–∏—è]
class ColoredFormatter(logging.Formatter):
    """–¶–≤–µ—Ç–Ω–æ–π —Ñ–æ—Ä–º–∞—Ç—Ç–µ—Ä –¥–ª—è –≤—ã–≤–æ–¥–∞ –ª–æ–≥–æ–≤ –≤ –∫–æ–Ω—Å–æ–ª—å.

    –ö–ª–∞—Å—Å —Ä–∞—Å—à–∏—Ä—è–µ—Ç —Å—Ç–∞–Ω–¥–∞—Ä—Ç–Ω—ã–π `logging.Formatter`, –¥–æ–±–∞–≤–ª—è—è —Ü–≤–µ—Ç–æ–≤–æ–µ –æ—Ñ–æ—Ä–º–ª–µ–Ω–∏–µ
    –≤ –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –æ—Ç —É—Ä–æ–≤–Ω—è –ª–æ–≥–∏—Ä–æ–≤–∞–Ω–∏—è. –ü–æ–¥–¥–µ—Ä–∂–∏–≤–∞—é—Ç—Å—è —É—Ä–æ–≤–Ω–∏ DEBUG, INFO,
    WARNING, ERROR –∏ CRITICAL.

    Attributes:
        COLORS (dict): –°–æ–ø–æ—Å—Ç–∞–≤–ª–µ–Ω–∏–µ —É—Ä–æ–≤–Ω–µ–π –ª–æ–≥–∏—Ä–æ–≤–∞–Ω–∏—è —Å —Ü–≤–µ—Ç–∞–º–∏ –∏–∑ –±–∏–±–ª–∏–æ—Ç–µ–∫–∏ colorama.
    """

    COLORS = {
        "DEBUG": Fore.CYAN,
        "INFO": Fore.GREEN,
        "WARNING": Fore.YELLOW,
        "ERROR": Fore.RED,
        "CRITICAL": Back.RED + Fore.WHITE + Style.BRIGHT,
    }



[–¥–æ–∫—É–º–µ–Ω—Ç–∞—Ü–∏—è]
    def format(self, record):
        """–§–æ—Ä–º–∞—Ç–∏—Ä—É–µ—Ç –∑–∞–ø–∏—Å—å –ª–æ–≥–∞, –ø—Ä–∏–º–µ–Ω—è—è —Ü–≤–µ—Ç –≤ –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –æ—Ç —É—Ä–æ–≤–Ω—è.

        –¶–≤–µ—Ç —É—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ—Ç—Å—è –ø–µ—Ä–µ–¥ –≤—ã–∑–æ–≤–æ–º –±–∞–∑–æ–≤–æ–≥–æ –º–µ—Ç–æ–¥–∞ `format()` —Ä–æ–¥–∏—Ç–µ–ª—å—Å–∫–æ–≥–æ –∫–ª–∞—Å—Å–∞.

        Args:
            record (logging.LogRecord): –û–±—ä–µ–∫—Ç –∑–∞–ø–∏—Å–∏ –ª–æ–≥–∞.

        Returns:
            str: –û—Ç—Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–Ω–∞—è —Å—Ç—Ä–æ–∫–∞ –ª–æ–≥–∞ —Å —Ü–≤–µ—Ç–æ–≤—ã–º –≤—ã–¥–µ–ª–µ–Ω–∏–µ–º.
        """

        log_level = record.levelname
        color = self.COLORS.get(log_level, "")
        self._style._fmt = f"{color}{LOG_FORMAT}{Style.RESET_ALL}"
        return super().format(record)








[–¥–æ–∫—É–º–µ–Ω—Ç–∞—Ü–∏—è]
def get_logger(name=None):
    """–í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –Ω–∞—Å—Ç—Ä–æ–µ–Ω–Ω—ã–π –ª–æ–≥–≥–µ—Ä —Å —É–∫–∞–∑–∞–Ω–Ω—ã–º –∏–º–µ–Ω–µ–º.

    –°–æ–∑–¥–∞—ë—Ç –∏ –Ω–∞—Å—Ç—Ä–∞–∏–≤–∞–µ—Ç –ª–æ–≥–≥–µ—Ä, –∫–æ—Ç–æ—Ä—ã–π –≤—ã–≤–æ–¥–∏—Ç —Å–æ–æ–±—â–µ–Ω–∏—è –≤ –∫–æ–Ω—Å–æ–ª—å —Å —Ü–≤–µ—Ç–æ–≤–æ–π
    –∏–Ω–¥–∏–∫–∞—Ü–∏–µ–π —É—Ä–æ–≤–Ω—è –ª–æ–≥–∏—Ä–æ–≤–∞–Ω–∏—è –∏ –∑–∞–ø–∏—Å—ã–≤–∞–µ—Ç –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏–æ–Ω–Ω—ã–µ –∏ –±–æ–ª–µ–µ —Å–µ—Ä—å—ë–∑–Ω—ã–µ
    —Å–æ–æ–±—â–µ–Ω–∏—è –≤ —Ñ–∞–π–ª.

    Args:
        name (str, optional): –ò–º—è –ª–æ–≥–≥–µ—Ä–∞. –ï—Å–ª–∏ –Ω–µ —É–∫–∞–∑–∞–Ω–æ, –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç—Å—è –∫–æ—Ä–Ω–µ–≤–æ–π –ª–æ–≥–≥–µ—Ä.

    Returns:
        logging.Logger: –ù–∞—Å—Ç—Ä–æ–µ–Ω–Ω—ã–π —ç–∫–∑–µ–º–ø–ª—è—Ä –ª–æ–≥–≥–µ—Ä–∞.
    """

    # –ù–∞—Å—Ç—Ä–æ–π–∫–∞ –∫–æ—Ä–Ω–µ–≤–æ–≥–æ –ª–æ–≥–≥–µ—Ä–∞ –ø—Ä–∏ –ø–æ–º–æ—â–∏ configs/config.py
    logger = logging.getLogger(name)
    logger.setLevel(Config.LOG_LEVEL)

    # –ü—Ä–µ–¥–æ—Ç–≤—Ä–∞—â–∞–µ–º –¥—É–±–ª–∏—Ä–æ–≤–∞–Ω–∏–µ –ª–æ–≥–æ–≤ (–µ—Å–ª–∏ —É–∂–µ –µ—Å—Ç—å —Ö—ç–Ω–¥–ª–µ—Ä—ã)
    if logger.hasHandlers():
        return logger

    # --- –ö–æ–Ω—Å–æ–ª—å–Ω—ã–π –≤—ã–≤–æ–¥ ---
    console_handler = logging.StreamHandler()
    console_handler.setLevel(logging.DEBUG)
    console_formatter = ColoredFormatter(fmt=LOG_FORMAT, datefmt=DATE_FORMAT)
    console_handler.setFormatter(console_formatter)
    logger.addHandler(console_handler)

    # --- –õ–æ–≥–∏—Ä–æ–≤–∞–Ω–∏–µ –≤ —Ñ–∞–π–ª ---
    log_file = os.path.join(LOGS_DIR, f"{datetime.now().strftime('%Y-%m-%d')}.log")
    file_handler = logging.FileHandler(log_file, encoding="utf-8")
    file_handler.setLevel(logging.INFO)
    file_formatter = logging.Formatter(fmt=LOG_FORMAT, datefmt=DATE_FORMAT)
    file_handler.setFormatter(file_formatter)
    logger.addHandler(file_handler)

    return logger
